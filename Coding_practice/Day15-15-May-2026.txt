
---
- name: Install Java on Pathnex server
  hosts: all
  become: yes

  tasks:
    - name: Install OpenJDK
      yum:
        name: java-11-openjdk
        state: present

    - name: Set JAVA_HOME
      lineinfile:
        path: /etc/profile
        line: 'export JAVA_HOME=/usr/lib/jvm/java-11-openjdk'



provider "aws" {
  region = "us-east-1"
}
resource "aws_s3_bucket" "PathnexBucket" {
  bucket = "pathnex-devops-bucket"

  tags = {
    Name = "PathnexBucket"
  }
}

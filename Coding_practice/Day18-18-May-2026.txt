  name: pathnex-affinity

  nodeAffinity:
   requiredDuringSchedulingIgnoredDuringExecution:
      nodeSelectorTerms:
      - matchExpressions:
              - key: disktype

     - ssd
  containers:
    - name: web
      image: nginx
